class Solution {
    public int maxArea(int[] height) {
        int i = 0;
        int j = height.length - 1;
        int maxArea = Math.min(height[i], height[j]) * (j - i);
        
        while (i < j) {
            if (height[i] > height[j]) {
                j--;
            } else {
                i++;
            }
            maxArea = Math.max(maxArea, Math.min(height[i], height[j]) * (j - i));
        }
        
        return maxArea;
    }
}
