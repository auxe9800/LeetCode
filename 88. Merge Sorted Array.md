class Solution {
    public void merge(int[] nums1, int m, int[] nums2, int n) {
        if (nums1.length == 0 || nums2.length == 0) {
            return;
        }
        int i = 0;
        int j = 0;
        
        while (i < m + n) {
            if (nums1[i] == 0 && i >= m) {
                nums1[i] = nums2[j];
                j++;
            } else if (nums1[i] > nums2[j]) {
                int temp = nums1[i];
                nums1[i] = nums2[j];
                nums2[j] = temp;
                Arrays.sort(nums2);
            }
            i++;
        }
    }
}
