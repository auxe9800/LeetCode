/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {
    public ListNode getIntersectionNode(ListNode headA, ListNode headB) {
        ListNode a = headA;
        ListNode b = headB;
        int count = 0;
        
        while (count <= 2) {
            if (a == b) {
                return a;
            }
            a = a.next;
            b = b.next;
            if (a == null) {
                a = headB;
                count++;
            }
            if (b == null) {
                b = headA;
                count++;
            }
        }
        
        return null;
    }
}
