class Solution {
    public int[] twoSum(int[] numbers, int target) {
        int[] result = new int[2];
        int left = 0;
        int right = numbers.length - 1;
        
        while (left <= right) {
            if (target - numbers[left] - numbers[right] == 0) {
                result[0] = left + 1;
                result[1] = right + 1;
                return result;
            } else if (target - numbers[left] - numbers[right] > 0) {
                left++;
            } else {
                right--;
            }
        }
        return result;
    }
}
