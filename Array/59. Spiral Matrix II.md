class Solution {
    public int[][] generateMatrix(int n) {
        int[][] ans = new int[n][n];
        int startX = 0;
        int startY = 0;
        int loop = n/2;
        int offset = 1;
        int mid = n/2;
        int count = 1;
        
        while (loop > 0) {
            int i = startX;
            int j = startY;
            
            for(;j < startY + n - offset;j++) {
                ans[startX][j] = count++;
            }
            for(;i < startX + n - offset;i++) {
                ans[i][j] = count++;
            }
            for(;j > startY;j--) {
                ans[i][j] = count++;
            }
            for(;i > startX;i--) {
                ans[i][j] = count++;
            }
            loop--;
            startX++;
            startY++;
            offset+=2;
        }
        if (n%2 == 1) {
            ans[mid][mid] = count;
        }
        
        return ans;
    }
}
