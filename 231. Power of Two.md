class Solution {
    public boolean isPowerOfTwo(int n) {
        int x = 0;
        
        while (n >= Math.pow(2, x)) {
            int power = (int) Math.pow(2, x);
            if (n == power) {
                return true;
            } 
            x++;
        }
        
        return false;
    }
}
