class Solution {
    public int jump(int[] nums) {
        int cur = 0;
        int next = 0;
        int ans = 0;
        
        for (int i = 0;i < nums.length - 1;i++) {
            next = Math.max(nums[i] + i, next);
            
            if (i == cur) {
                ans++;
                cur = next;
            }
        }
        
        return ans;
    }
}
