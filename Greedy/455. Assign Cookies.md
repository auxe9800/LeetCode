class Solution {
    public int findContentChildren(int[] g, int[] s) {
        int result = 0, i = 0, j = 0;
        Arrays.sort(g);
        Arrays.sort(s);
        
        while (i < s.length && j < g.length) {
            if (s[i] >= g[j]) {
                result++;
                i++;
                j++;
            } else {
                i++;
            }
        }
        
        return result;
    }
}
