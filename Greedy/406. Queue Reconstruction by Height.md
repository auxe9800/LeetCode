class Solution {
    public int[][] reconstructQueue(int[][] people) {
        Arrays.sort(people, (a, b) -> {
            if (a[0] == b[0]) {
                return a[1] - b[1];
            }
            return b[0] - a[0];
        });
        
        List<int[]> queue = new LinkedList<>();
        
        for (int i = 0;i < people.length;i++) {
            queue.add(people[i][1], people[i]);
        }
        
        return queue.toArray(new int[people.length][]);
    }
}
