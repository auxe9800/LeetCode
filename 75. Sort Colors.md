class Solution {
    public void sortColors(int[] nums) {
        int _0 = 0;
        int _1 = 0;
        
        for (int i = 0;i < nums.length;i++) {
            if (nums[i] == 0) {
                _0++;
            } else if (nums[i] == 1) {
                _1++;
            }
        }
        for (int i = 0;i < nums.length;i++) {
            if (_0 > 0) {
                nums[i] = 0;
                _0--;
            } else if (_1 > 0) {
                nums[i] = 1;
                _1--;
            } else {
                nums[i] = 2;
            }
        }
    }
}
