class MyStack {
    Queue<Integer> q1;
    Queue<Integer> q2;

    public MyStack() {
        this.q1 = new LinkedList<Integer>();
        this.q2 = new LinkedList<Integer>();
    }
    
    public void push(int x) {
        if (this.q1.isEmpty() && this.q2.isEmpty()) {
            this.q1.offer(x);
        } else if (!this.q1.isEmpty()) {
            this.q1.offer(x);
        } else {
            this.q2.offer(x);
        }
    }
    
    public int pop() {
        int target = 0;
        while (!this.q1.isEmpty()) {
            target = this.q1.poll();
            if (this.q1.isEmpty()) {
                return target;
            }
            this.q2.offer(target);
        }
        while (!this.q2.isEmpty()) {
            target = this.q2.poll();
            if (this.q2.isEmpty()) {
                return target;
            }
            this.q1.offer(target);
        }
        
        return target;
    }
    
    public int top() {
        int target = 0;
        while (!this.q1.isEmpty()) {
            target = this.q1.poll();
            this.q2.offer(target);
            if (this.q1.isEmpty()) {
                return target;
            }
        }
        while (!this.q2.isEmpty()) {
            target = this.q2.poll();
            this.q1.offer(target);
            if (this.q2.isEmpty()) {
                return target;
            }
        }
        
        return target;
        
    }
    
    public boolean empty() {
        return this.q1.isEmpty() && this.q2.isEmpty();
    }
}
