class Solution {
    public String removeDuplicates(String s) {
        Stack<Character> myStack = new Stack<Character>();
        StringBuilder sb = new StringBuilder();
        
        for (int i = 0;i < s.length();i++) {
            char ch = s.charAt(i);
            if (!myStack.empty() && myStack.peek() == ch) {
                myStack.pop();
            } else {
                myStack.push(ch);
            }
        }
        while (!myStack.empty()) {
            sb.insert(0, myStack.pop());
        }
        
        return sb.toString();
    }
}
