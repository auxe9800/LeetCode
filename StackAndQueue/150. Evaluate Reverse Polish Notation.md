class Solution {
    public int evalRPN(String[] tokens) {
        Stack<Integer> myStack = new Stack<Integer>();
        
        for (String t : tokens) {
            if (t.equals("+")) {
                int last = myStack.pop();
                int fast = myStack.pop();
                myStack.push(fast+last);
            } else if (t.equals("-")) {
                int last = myStack.pop();
                int fast = myStack.pop();
                myStack.push(fast-last);
            } else if (t.equals("*")) {
                int last = myStack.pop();
                int fast = myStack.pop();
                myStack.push(fast*last);
            } else if (t.equals("/")) {
                int last = myStack.pop();
                int fast = myStack.pop();
                myStack.push(fast/last);
            } else {
                myStack.push(Integer.valueOf(t));
            }
        }
        
        return myStack.pop();
    }
}
