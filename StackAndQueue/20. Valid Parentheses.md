class Solution {
    public boolean isValid(String s) {
        Stack<Character> myStack = new Stack<Character>();
        
        for (int i = 0;i < s.length();i++) {
            char temp = s.charAt(i);
            
            if (temp == '(') {
                myStack.push(')');
            } else if (temp == '[') {
                myStack.push(']');
            } else if (temp == '{') {
                myStack.push('}');
            } else if (myStack.empty() || myStack.peek() != temp) {
                return false;
            } else {
                myStack.pop();
            }
        }
        
        return myStack.empty();
    }
}
