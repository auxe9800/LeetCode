class Solution {
    public int orangesRotting(int[][] grid) {
        int minutes = 0;
        int freshOrange = 0;
        int m = grid.length;
        int n = grid[0].length;
        Queue<int[]> queue = new LinkedList<>();
        
        for (int i = 0;i < m;i++) {
            for (int j = 0;j < n;j++) {
                if (grid[i][j] == 1) {
                    freshOrange++;
                } else if (grid[i][j] == 2) {
                    queue.offer(new int[] {i, j});
                }
            }
        }
        if (freshOrange == 0) {
            return minutes;
        }
        
        int[][] dirs = new int[][] {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
        
        while (!queue.isEmpty()) {
            int size = queue.size();
            
            while (size > 0) {
                int[] cell = queue.poll();
                size--;

                for (int[] d : dirs) {
                    int row = cell[0] + d[0];
                    int col = cell[1] + d[1];

                    if (row < 0 || col < 0 || row >= m || col >= n || grid[row][col] == 0 || grid[row][col] == 2) {
                        continue;
                    }
                    grid[row][col] = 2;
                    queue.offer(new int[] {row, col});
                    freshOrange--;
                }
            }
            minutes++;
            if (freshOrange == 0) {
                return minutes;
            }
        }
        
        return -1;
    }
}
