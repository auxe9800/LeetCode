class Solution {
    public String reverseWords(String s) {
        String[] sa = s.split(" ");
        
        for (int i = 0;i < sa.length;i++) {
            sa[i] = reverseWord(sa[i]);
        }
        
        return String.join(" ", sa);
    }
    public String reverseWord(String s) {
        char[] ca = s.toCharArray();
        int left = 0;
        int right = ca.length - 1;
        
        while (left <= right) {
            char temp = ca[left];
            ca[left] = ca[right];
            ca[right] = temp;
            
            left++;
            right--;
        }
        
        return new String(ca);
    }
}
