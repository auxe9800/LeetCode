class Solution {
    public int maxSubArray(int[] nums) {
        int[] dp = new int[nums.length];
        int max = Integer.MIN_VALUE;
        
        dp[0] = nums[0];
        for (int i = 1;i < dp.length;i++) {
            dp[i] = Math.max(nums[i], dp[i - 1] + nums[i]);
        }
        for (int i = 0;i < dp.length;i++) {
            max = Math.max(max, dp[i]);
        }
        
        return max;
    }
}
