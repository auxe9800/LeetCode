class Solution {
    public boolean isHappy(int n) {
        Set<Integer> myHashSet = new HashSet<Integer>();
        int sum = 0;
        int temp = n;
        
        while (sum != 1) {
            sum = 0;
            while (temp != 0) {
                int t = temp % 10;
                sum += t*t;
                temp = (int) temp / 10;
            }
            if (myHashSet.contains(sum)) {
                return false;
            } else {
                myHashSet.add(sum);
                temp = sum;
            }
        }
        
        return true;
    }
}
