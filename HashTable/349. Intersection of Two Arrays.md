class Solution {
    public int[] intersection(int[] nums1, int[] nums2) {
        Set<Integer> myHashSet = new HashSet<Integer>();
        Set<Integer> ans = new HashSet<Integer>();
        
        for (int i : nums1) {
            myHashSet.add(i);
        }
        for (int i : nums2) {
            if (myHashSet.contains(i)) {
                ans.add(i);
            }
        }
        return ans.stream().mapToInt(Number::intValue).toArray();
    }
}
