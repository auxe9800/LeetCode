class Solution {
    public TreeNode convertBST(TreeNode root) {
        traversal(root, 0);
        return root;
    }
    public int traversal(TreeNode cur, int sum) {
        if (cur == null) {
            return sum;
        }
        int rightSum = traversal(cur.right, sum);
        cur.val += rightSum;
        int leftSum = traversal(cur.left, cur.val);
        return leftSum;
    }
}
