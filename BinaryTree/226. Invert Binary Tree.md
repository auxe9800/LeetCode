class Solution {
    public TreeNode invertTree(TreeNode root) {
        if (root == null) {
            return root;
        }
        Queue<TreeNode> queue = new LinkedList<>();
        queue.offer(root);
        
        while (!queue.isEmpty()) {
            int size = queue.size();
            
            while (size > 0) {
                TreeNode tmp = queue.poll();
                TreeNode swap = tmp.left;
                tmp.left = tmp.right;
                tmp.right = swap;
                if (tmp.left != null) {
                    queue.offer(tmp.left);
                }
                if (tmp.right != null) {
                    queue.offer(tmp.right);
                }
                size--;
            }
        }
        
        return root;
    }
}
