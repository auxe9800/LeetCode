class Solution {
    public Node connect(Node root) {
        Queue<Node> queue = new LinkedList<>();
        if (root == null) {
            return null;
        }
        
        queue.offer(root);
        while (!queue.isEmpty()) {
            int len = queue.size();
            
            while (len > 0) {
                Node temp = queue.poll();
                temp.next = queue.peek();
                if (len == 1) {
                    temp.next = null;
                }
                
                if (temp.left != null) {
                    queue.offer(temp.left);
                }
                if (temp.right != null) {
                    queue.offer(temp.right);
                }
                len--;
            }
        }
        
        return root;
    }
}
