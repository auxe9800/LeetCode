/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int val = 0;
    int count = 0;
    
    public int kthSmallest(TreeNode root, int k) {
        count = k;
        traversal(root);
        
        return val;
    }
    public void traversal(TreeNode cur) {
        if (cur.left != null) {
            traversal(cur.left);
        }
        count--;
        if (count == 0) {
            val = cur.val;
        }
        if (cur.right != null) {
            traversal(cur.right);
        }
    }
}
