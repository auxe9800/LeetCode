class Solution {
    public int maxDepth(Node root) {
        if (root == null) {
            return 0;
        }
        int depth = 0;
        Queue<Node> queue = new LinkedList<>();
        queue.offer(root);
        
        while (!queue.isEmpty()) {
            depth++;
            int size = queue.size();
            while (size > 0) {
                Node temp = queue.poll();
                
                for (Node n : temp.children) {
                    queue.offer(n);
                }
                size--;
            }
        }
        
        return depth;
    }
}
