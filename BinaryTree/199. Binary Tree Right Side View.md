class Solution {
    public List<Integer> rightSideView(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        Deque<TreeNode> deque = new LinkedList<>();
        if (root == null) {
            return result;
        }
        
        deque.offer(root);
        while (!deque.isEmpty()) {
            int size = deque.size();
            
            while (size > 0) {
                TreeNode temp = deque.poll();
                if (size == 1) {
                    result.add(temp.val);
                }
                if (temp.left != null) {
                    deque.offer(temp.left);
                }
                if (temp.right != null) {
                    deque.offer(temp.right);
                }
                size--;
            }
        }
        
        return result;
    }
}
