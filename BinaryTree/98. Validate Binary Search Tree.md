class Solution {
    TreeNode max;
    public boolean isValidBST(TreeNode root) {
        if (root == null) {
            return true;
        }
        boolean leftSubTree = isValidBST(root.left);
        if (!leftSubTree) {
            return false;
        }
        
        if (max != null && root.val <= max.val) {
            return false;
        }
        max = root;
        
        boolean rightSubTree = isValidBST(root.right);
        if (!rightSubTree) {
            return false;
        }
        
        return true;
    }
}
