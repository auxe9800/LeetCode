class Solution {
    public List<List<Integer>> levelOrder(Node root) {
        List<List<Integer>> results = new ArrayList<>();
        Queue<Node> queue = new LinkedList<>();
        
        if (root == null) {
            return results;
        }
        queue.offer(root);
        while (!queue.isEmpty()) {
            int len = queue.size();
            List<Integer> result = new ArrayList<>();
            while (len > 0) {
                Node temp = queue.poll();
                result.add(temp.val);
                
                for (Node n : temp.children) {
                    queue.offer(n);
                }
                len--;
            }
            results.add(result);
        }
        
        return results;
    }
}
