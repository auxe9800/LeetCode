class Solution {
    public int findBottomLeftValue(TreeNode root) {
        int leftMost = root.val;
        Queue<TreeNode> queue = new LinkedList<>();
        queue.offer(root);
        
        while (!queue.isEmpty()) {
            int size = queue.size();
            int left = size;
            
            while (size > 0) {
                TreeNode cur = queue.poll();
                if (size == left) {
                    leftMost = cur.val;
                }
                if (cur.left != null) {
                    queue.offer(cur.left);
                }
                if (cur.right != null) {
                    queue.offer(cur.right);
                }
                size--;
            }
        }
        
        return leftMost;
    }
}
