class Solution {
    public int countNodes(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int leftDepth = 0;
        int rightDepth = 0;
        
        TreeNode temp = root;
        while (temp.left != null) {
            leftDepth++;
            temp = temp.left;
        }
        temp = root;
        while (temp.right != null) {
            rightDepth++;
            temp = temp.right;
        }
        if (rightDepth == leftDepth) {
            return (2 << leftDepth) - 1;
        }
        
        return countNodes(root.left) + countNodes(root.right) + 1;
    }
}
